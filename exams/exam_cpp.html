Part 6

  • Class Point: Implement class Point. Allow the following operations. Be strict about where you implement functions (members or globals), about code reuse, etc.
    Point<int> p1(5,6);
    Point<int> p2(12,18);
    Point<int> p3;
    p3 = p1 + p2;
    Point<int> p4 = p3 - p1;
    Point<int> p5(0,0);
    assert(p4==p2);
    Point<int> p6(p1);
    p6 += 1;
    assert(p6 == Point<int>(6,7));
    p2 /= 2;
    assert(p2 == Point<int>(6,9));
